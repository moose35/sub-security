Javascript Jargon

        . - period, dot
        () - parenthesis
        [] - brackets, square brackets, square brackets
        {} - braces, curlies, curly braces, curly brackets
        < - less than, angle bracket(html)
        > - greater than, angle bracket, gator
        / - slash, forward slash, whack(Dev lectures)
        \ - slash, back slash, back whack(Dev lectures)
        ! - exclamation point, bang, not
        # - hashtag, pound, number sign
        * - asterisk, star, splat
        || - or, pipes, vertical bar
        && - and, ampersand
        $ - dollar sign, bling, cash
        NaN - not a number
